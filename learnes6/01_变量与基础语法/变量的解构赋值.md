# 解构赋值

## 数组解构

```js
// 基本语法
let [a, b, c] = [1, 2, 3]

// 嵌套解构
let [foo, [[bar], baz]] = [1, [[2], 3]]

// 不完全解构
let [x, y] = [1, 2, 3]

// 默认值 生效条件：数组中的值 === undefined
var {x, y = 5} = {x: 1}
x // 1
y // 5
```

## 对象解构

与数组解构不同点：变量必须与属性**同名**

```js
// 基本语法 解构失败，变量值为 undefined
let {foo, bar} = {foo: 'aaa', bar: 'bbb'}

// 重命名 foo 是要匹配的属性名 baz 是将要创建的变量名
let {foo: baz} = {foo: 'aaa', bar: 'bbb'}

// 嵌套解构
const node = {
  loc: {
    start: {
      line: 1,
      column: 5
    }
  }
}
let {
  loc,
  loc: {start},
  loc: {
    start: {line}
  }
} = node

// 默认值 生效条件：属性的值 === undefined
var {x, y = 5} = {x: 1}
```

**特殊情况**：

```js
// 1. 已经声明的变量用于解构，需要加上()
let x
// ({x} = {x: 1})
// 2. 数组的解构可以使用对象解构的方式
let {0: first, [arr.length - 1]: last} = [1, 2, 3]
```

## 其他解构

解构赋值的规则：只要等号右边不是对象或者数组，会先去转换成对象

```js
// 字符串解构
const [a, b, c, d, e] = 'hello' // 字符串 -> 类似数组的对象
let {length: len} = 'hello'

// Number 和 Boolean
let {toString: s} = 123
s === Number.prototype.toString // true

let {toString: s} = true
s === Boolean.prototype.toString // true

// 函数参数解构
function add([x, y]) {
  return x + y
}

add([1, 2]) // 3

// 函数参数解构的默认值
function move({x = 0, y = 0} = {}) {
  return [x, y]
}
move({x: 3, y: 8}) // [3, 8]
move({x: 3}) // [3, undefined]
move({}) // [undefined, undefined]
move() // [0, 0]
```

## 使用场景

- **交换变量**

```js
let x = 1
let y = 2
// 不需要临时变量
;[x, y] = [y, x]
```

- **函数参数收集**

```js
// 参数是一组有次序的值
function f([x, y, z]) { ... }
f([1, 2, 3]);

// 参数是一组无次序的值
function f({x, y, z}) { ... }
f({z: 3, y: 2, x: 1});
```

- **按需提取数据**

```js
// 按需提取函数返回值
let {foo, bar} = example()
// 按需提取对象属性
let {id, status, data: number} = Data
// 按需提取模块方法
const {SourceMapConsumer, SourceNode} = require('source-map')
```
